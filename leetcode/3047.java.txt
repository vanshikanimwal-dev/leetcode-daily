class Solution {
    public long largestSquareArea(int[][] bottomLeft, int[][] topRight) {
        int maxSide = 0;

        int n = bottomLeft.length;
        for (int i = 0; i < n; ++i) {
            int ax1 = bottomLeft[i][0];
            int ay1 = bottomLeft[i][1];
            int ax2 = topRight[i][0];
            int ay2 = topRight[i][1];

            for (int j = i + 1; j < n; ++j) {
                int bx1 = bottomLeft[j][0];
                int by1 = bottomLeft[j][1];
                int bx2 = topRight[j][0];
                int by2 = topRight[j][1];

                // Overlap width
                int overlapWidth = Math.min(ax2, bx2) - Math.max(ax1, bx1);
                // Overlap height
                int overlapHeight = Math.min(ay2, by2) - Math.max(ay1, by1);

                // If both positive, they overlap
                if (overlapWidth > 0 && overlapHeight > 0) {
                    int side = Math.min(overlapWidth, overlapHeight);
                    maxSide = Math.max(maxSide, side);
                }
            }
        }

        // Area = side^2
        return (long) maxSide * maxSide;
    }
}
