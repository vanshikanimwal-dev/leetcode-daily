class Solution {
    public boolean judgeCircle(String moves) {
        
        int x = 0; // horizontal (L,R)
        int y = 0; // vertical (U,D)
        
        for (char ch : moves.toCharArray()) {
            
            if (ch == 'U') y++;
            else if (ch == 'D') y--;
            else if (ch == 'L') x--;
            else if (ch == 'R') x++;
        }
        
        // check if back to origin
        return x == 0 && y == 0;
    }
}
