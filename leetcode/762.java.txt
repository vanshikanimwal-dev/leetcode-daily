class Solution {

    public int countPrimeSetBits(int left, int right) {

        int count = 0;

        for (int num = left; num <= right; num++) {

            int bits = Integer.bitCount(num); // count 1s

            if (isPrime(bits)) {
                count++;
            }
        }

        return count;
    }

    private boolean isPrime(int x) {
        if (x < 2) return false;

        for (int i = 2; i * i <= x; i++) {
            if (x % i == 0) return false;
        }

        return true;
    }
}
